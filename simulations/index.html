Cette page présente la planification de simulations d’exécution d’algorithmes de détection de changement via le programme présenté sur la page Détection de changement, ainsi que les résultats obtenus en termes de performances, temps de calcul, cosommation CPU et mémoire, température au sein de la machine…

Plan de simulation

Hyperparamètres

Hyperparamètres Valeur
Taille des images (px) 500x500 - 500x1000 - 1000x1000
Nombre d’images 4 - 17 - 68
Taille de la fenêtre de calcul de covariance (px) 5x5 - 11x11 - 21x21
Nombre de threads pour calcul de covariance 5 - 10 - 12

Nombre de types de simulations : $81$ ($\times 10$ afin d’obtenir des descriptions statistiques des résultats)

Format du résultat

  • Temps d’exécution (ns)
  • Données sur la période d’exécution (format csv)

Fichier d’exécution

 1#!/bin/bash
 2
 3while getopts 'f:p:P' OPTION; do
 4  case "$OPTION" in
 5    f)
 6      name_file="$OPTARG" #récupération du nom du fichier python à exécuter
 7      ;;
 8    P) 
 9      by_pid=true
10      get_pid=true
11      ;;
12    p) 
13      by_pid=true
14      get_pid=false
15      npid="$OPTARG"
16      ;;
17    \?)
18      echo "Invalid option: $OPTARG" 1>&2
19      exit 1
20      ;;
21  esac
22done
23
24: ${name_file:?Missing -f} #vérification de l'argument du nom du fichier python
25
26for name_image in "500x500x4" "1000x1000x4" "500x500x17" "1000x1000x17" "500x500x68" "1000x1000x68"; do  #nom du repository avec les images redimensionnées
27  for window_size in 5 11 21; do 
28    for n_jobs_cov in 5 10 12; do
29      for i in {1..10}; do  #réplication de l'expérience 10 fois
30        t1=$(date -u +%Y-%m-%dT%T.%9NZ)
31        echo "*************************************************"
32        echo "Time start: $t1"
33        echo "*************************************************"
34        echo "Running python script: $name_file"
35        echo "*************************************************"
36
37        python3 $name_file $name_image $window_size $n_jobs_cov  #exécution du fichier python avec les hyperparamètres
38        t2=$(date -u +%Y-%m-%dT%T.%9NZ)
39
40        echo "*************************************************"
41        echo "Time stop: $t2"
42        echo "*************************************************"
43
44        echo "$name_image,$window_size,$n_jobs_cov,$(($(date -d "$t2" +%s%9N) - $(date -d "$t1" +%s%9N)))" >> execution_time  #sauvegarde du temps d'exécution
45
46        if [ "$by_pid" = true ]; then
47          if [ "$get_pid" = true ]; then
48            npid=$(cat python_process.pid)
49          fi
50          echo "Process ID: ${npid}"
51          query="data=from(bucket: \"telegraf_bucket\")
52            |> range(start: ${t1}, stop: ${t2})
53            |> filter(fn: (r) => r[\"_measurement\"] == \"procstat\")
54            |> filter(fn: (r) => r[\"pid\"] == \"${npid}\")
55            |> aggregateWindow(every: 1s, fn: mean, createEmpty: false)
56            |> yield(name: \"mean\")"
57        else
58          query="data=from(bucket: \"telegraf_bucket\")
59            |> range(start: ${t1}, stop: ${t2})
60            |> aggregateWindow(every: 1s, fn: mean, createEmpty: false)
61            |> yield(name: \"mean\")"
62        fi
63
64        echo $query > query
65
66        sudo docker cp query influxdb:/query
67        sudo docker exec -it influxdb sh -c 'influx query -f query -r' > metrics_output
68        sudo docker exec -it influxdb sh -c 'rm query'
69        sudo rm query
70
71        mv metrics_output $name_image.$window_size.$n_jobs_cov.$i.metrics_output
72        echo "*************************************************"
73        echo "File $name_image.$window_size.$n_jobs_cov.$i.metrics_output created"
74        echo "*************************************************"
75        mv $name_image.$window_size.$n_jobs_cov.$i.metrics_output results_simulation/  #sauvegarde du résultat
76      done
77    done
78  done
79done
